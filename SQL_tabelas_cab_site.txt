CREATE TABLE cliente (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100),
    cpf_cnpj VARCHAR(20) NOT NULL,
    telefone VARCHAR(20),
    endereco VARCHAR(200)
);

CREATE TABLE produto (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100),
    tipo VARCHAR(50) NOT NULL,
    preco DOUBLE PRECISION,
    estoque DOUBLE PRECISION
);

CREATE TABLE pedido (
    id SERIAL PRIMARY KEY,
    cliente_id INTEGER NOT NULL,
    data_pedido DATE NOT NULL,
    status VARCHAR(20),
    CONSTRAINT fk_pedido_cliente_id FOREIGN KEY (cliente_id) REFERENCES cliente(id)
);

CREATE TABLE itens_pedido (
    id SERIAL PRIMARY KEY,
    pedido_id INTEGER NOT NULL,
    produto_id INTEGER NOT NULL,
    quantidade DOUBLE PRECISION,
    preco_unitario DOUBLE PRECISION,
    CONSTRAINT fk_itens_pedido_id FOREIGN KEY (pedido_id) REFERENCES pedido(id),
    CONSTRAINT fk_itens_produto_id FOREIGN KEY (produto_id) REFERENCES produto(id)
);

CREATE TABLE aplicacao (
    id SERIAL PRIMARY KEY,
    pedido_id INTEGER NOT NULL,
    data_aplicacao DATE,
    local_aplicacao VARCHAR(200),
    status_aplicacao VARCHAR(20) DEFAULT 'agendada',
    CONSTRAINT fk_aplicacao_pedido_id FOREIGN KEY (pedido_id) REFERENCES pedido(id)
);
